bool_t GetReWritePossible(Device* dev)
{
        CDFeatures* cdf = dev->cdFeatures();
        DVDFeatures* dvdf = dev->dvdFeatures();
        BDFeatures* bdf = dev->bdFeatures();
        bool_t res = cdf->canWriteCDRW();
        res = res || dvdf->canWriteDVDMinusRW();
        res = res || dvdf->canWriteDVDRam();
        res = res || dvdf->canWriteDVDPlusRW();
        res = res || bdf->canWriteBDRE();
        return res;
}

bool_t GetWritePossible(Device* dev)
{
        CDFeatures* cdf = dev->cdFeatures();
        DVDFeatures* dvdf = dev->dvdFeatures();
        BDFeatures* bdf = dev->bdFeatures();
        bool_t res = cdf->canWriteCDR();
        res = res || dvdf->canWriteDVDMinusR();
        res = res || dvdf->canWriteDVDMinusRDL();
        res = res || dvdf->canWriteDVDMinusRW();
        res = res || dvdf->canWriteDVDRam();
        res = res || dvdf->canWriteDVDPlusR();
        res = res || dvdf->canWriteDVDPlusRDL();
        res = res || dvdf->canWriteDVDPlusRW();
        res = res || bdf->canWriteBDR();
        res = res || bdf->canWriteBDRE();
        return res;
}
